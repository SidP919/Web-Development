02 - Axios

//With Axios, fetching response from web server becomes quite easy
axios.get('https://api.cryptonator.com/api/ticker/btc-usd')
    .then(fetchedData => {
        console.log("ParsedResponseData: ");
        console.log(fetchedData);
        console.log("Price from ParsedResponseData: ");
        console.log(fetchedData.data.ticker.price);
    })
    .catch(err => {
        console.log(err);
    });
    //Please check console window/developer tools for Output of above code.
