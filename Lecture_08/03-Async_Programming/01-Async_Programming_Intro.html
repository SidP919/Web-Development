function fun() {
    console.log('Inside Fun');
}

//Creating a delay using a loop
function delay(n) {

    for (let i = 1; i <= n; i++){
        let x = new Date().getTime();
        while (new Date().getTime() < 1000 + x) { } 
    }
    fun();

}
