05 - Prototypes in JS


const person = {
        name: "Sidharth",
        age: 24,
        sayHello: function () {
            console.log(`Hello Every One, I\'m ${this.name}. I\'m ${this.age} years old.`);
        }
    }


let p1 = Object.create(person);
console.log(p1);
let p2 = Object.create(p1);
console.log(p2);
console.log(p1.sayHello());
console.log(p2.sayHello());
